    <script>
        // Tüm butonların genişliğini eşitle (2x2 grid için)
        function syncButtonWidths() {
            const heroActions = document.querySelector('.hero-actions');
            if (heroActions) {
                const containerWidth = heroActions.offsetWidth;
                const gap = 16; // 1rem = 16px
                const buttonWidth = (containerWidth - gap) / 2;
                
                const allButtons = [
                    heroActions.querySelector('.btn:nth-child(1)'),
                    heroActions.querySelector('.btn:nth-child(2)'),
                    heroActions.querySelector('.btn:nth-child(3)'),
                    heroActions.querySelector('.btn:nth-child(4)')
                ];
                
                // Tüm butonlara aynı genişliği uygula
                if (buttonWidth > 0) {
                    allButtons.forEach(btn => {
                        if (btn) {
                            btn.style.width = buttonWidth + 'px';
                            btn.style.minWidth = buttonWidth + 'px';
                            btn.style.maxWidth = buttonWidth + 'px';
                        }
                    });
                }
            }
        }
        
        document.addEventListener('DOMContentLoaded', function() {
            // Animasyonların tamamlanmasını bekle
            setTimeout(syncButtonWidths, 1000);
            // Ekstra kontrol için bir kez daha çalıştır
            setTimeout(syncButtonWidths, 1500);
        });
        
        // Pencere boyutu değiştiğinde tekrar hesapla
        let resizeTimeout;
        window.addEventListener('resize', function() {
            clearTimeout(resizeTimeout);
            resizeTimeout = setTimeout(syncButtonWidths, 200);
        });
    </script>
